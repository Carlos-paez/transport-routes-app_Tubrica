═══════════════════════════════════════════════════════════════════════════════
  ⚡ COMANDOS MÁS USADOS (REFERENCIA RÁPIDA)
═══════════════════════════════════════════════════════════════════════════════

Iniciar JavaScript:
  cd "javascript version" && docker-compose up -d

Iniciar PHP:
  cd "php version" && docker-compose up -d

Ver contenedores:
  docker ps

Ver logs:
  docker logs tubrica-javascript
  docker logs tubrica-php

Detener todo:
  docker stop tubrica-javascript tubrica-php

Reconstruir:
  docker-compose down && docker-compose build --no-cache && docker-compose up -d
